<script>
let students = [];

fetch('/WebDev_Repository/utils/get_students.php')
  .then(res => res.json())
  .then(data => {
    students = data.students;
    renderTable(students);
  });

function renderTable(data){
  const tbody = document.getElementById("studentTable");
  tbody.innerHTML = "";

  data.forEach(s => {
    tbody.innerHTML += `
      <tr>
        <td>${s.student_id}</td>
        <td>${s.student_name}</td>
        <td>${s.borrowed}</td>
        <td>${s.overdue}</td>
        <td class="${s.status === 'Active' ? 'status-active' : 'status-inactive'}">${s.status}</td>
        <td>
          <button class="action-btn ${s.status === 'Active' ? 'btn-suspend' : 'btn-activate'}"
            onclick="toggleStatus('${s.student_id}', '${s.status}')">
            ${s.status === 'Active' ? 'Suspend' : 'Activate'}
          </button>
        </td>
      </tr>
    `;
  });
}

document.getElementById("searchInput").addEventListener("input", e => {
  const val = e.target.value.toLowerCase();
  renderTable(
    students.filter(s =>
      s.student_id.toLowerCase().includes(val) ||
      s.student_name.toLowerCase().includes(val)
    )
  );
});

function toggleStatus(id, status){
  if(confirm("Are you sure you want to change this account status?")){
    // backend hook here
    alert("Status updated for " + id);
  }
}

function confirmLogout(e){
  e.preventDefault();
  if(confirm("Are you sure you want to logout?")){
    localStorage.clear();
    window.location.href = "http://localhost/WebDev_Repository/pages/sign_in_admin_librarian.html";
  }
}
